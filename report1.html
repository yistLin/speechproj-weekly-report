Use compute-mfcc-feats to compute mfcc from wav file and extract 13-dim vectors.


compute-mfcc-feats --verbose=2 \
	$options scp:material/$target.wav.scp \
	ark,t,scp:$path/$target.13.ark,$path/$target.13.scp 2> $log

02.extract.feat.sh

Add 1st order and 2nd order deltas to the 13-dim vectors to make it become 39-dim. Then, compute and apply CMVN on this 39-dim vectors.


add-deltas --delta-order=2 \
	ark:$path/$target.13.ark \
	ark,t:$path/$target.13.delta.ark
compute-cmvn-stats \
	ark:$path/$target.13.delta.ark ark:- | \
	apply-cmvn ark:- ark:$path/$target.13.delta.ark \
	ark:$path/$target.39.cmvn.ark 2> $log

gmm-init-mono

Initialize monophone GMM 


gmm-init-mono --binary=false $shared_phones_opt \
	"--train-feats=ark:subset-feats --n=10 \"ark,s,cs:$feat\" ark:- |" \
	train/topo $dim $dir/00.mdl $dir/tree 2> $log

compile-train-graphs

Creates training graphs 


compile-train-graphs $dir/tree $dir/00.mdl train/L.fst \
	ark:train/train.int ark:$dir/train.graph 2> $log

align-equal-compiled

Write an equally spaced alignment (first interation) 


align-equal-compiled ark:$dir/train.graph "ark,s,cs:$feat" \
	ark:$dir/00.ali 2> $log

gmm-align-compiled

Align features given [GMM-based] models 


gmm-align-compiled $scale_opts --beam=$beam --retry-beam=$[$beam*4] \
	$dir/$x.mdl ark:$dir/train.graph "ark,s,cs:$feat" ark:$dir/$x.ali 2> $log

gmm-acc-stats-ali

Accumulate stats for GMM training 


gmm-acc-stats-ali --binary=false $dir/00.mdl "ark,s,cs:$feat" \
  ark:$dir/train.ali $dir/$x.acc 2> $log

gmm-est

Do Maximum Likelihood re-estimation of GMM-based acoustic model 


gmm-est --binary=false --min-gaussian-occupancy=3 --mix-up=$numgauss \
  $dir/$x.mdl $dir/$x.acc $dir/$y.mdl 2> $log
